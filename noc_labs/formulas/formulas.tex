  $$C_{PID}^*(z)=K_p\Bigl(1+\frac{z}{T_i(z-1)}+\frac{z^2-2z+1}{T_d(z^2-z)}\Bigr)$$

%%file: img3 gain
\item Усилительное звено:
$$G^*(z)=K$$

%%file: img4 integral
\item Суммирующее звено:
$$G^*(z)=\frac{K}{z-d}, 0<d\le 1$$

%%file: img5 1st_order

$$G^*(z)=K\frac{z-d_1}{z-d_2}$$

$$G^*(z)=K\frac{z^2+d_1z+d_2}{z^2+d_1z+d_2}$$
